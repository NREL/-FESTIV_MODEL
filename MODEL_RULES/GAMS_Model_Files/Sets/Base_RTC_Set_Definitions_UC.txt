RTCSTART_GEN(GEN,INTERVAL) = YES$(RTCSTART(GEN,INTERVAL) EQ 1);  

RTCSHUT_GEN(GEN,INTERVAL) = YES$(RTCSHUT(GEN,INTERVAL) EQ 1);   

RTCPUMPSTART_GEN(GEN,INTERVAL) = YES$(RTCPUMPSTART(GEN,INTERVAL) EQ 1); 

RTCPUMPSHUT_GEN(GEN,INTERVAL) = YES$(RTCPUMPSHUT(GEN,INTERVAL) EQ 1);  

RTCCOMBINESTART_GEN(GEN,INTERVAL) =  RTCSTART_GEN(GEN,INTERVAL) + RTCPUMPSTART_GEN(GEN,INTERVAL);  

MIN_RUN_GEN(GEN,INTERVAL) = YES$(RTCSHUT(GEN,INTERVAL) EQ 1 AND GENVALUE(GEN,"MIN_RUN_TIME") GT INTERVAL_LENGTH);
MIN_RUN_GEN(GEN,INTERVAL) = MIN_RUN_GEN(GEN,INTERVAL) - OUTAGE_GEN(GEN);

MIN_DOWN_GEN(GEN,INTERVAL) = YES$((RTCSTART(GEN,INTERVAL) EQ 1 OR RTCPUMPSTART(GEN,INTERVAL) EQ 1) AND GENVALUE(GEN,"MIN_DOWN_TIME") GT INTERVAL_LENGTH);
MIN_DOWN_GEN(GEN,INTERVAL) = MIN_DOWN_GEN(GEN,INTERVAL) - OUTAGE_GEN(GEN);

=end
